//#target ps_2_0

// tell the compiler what permutations to build
//#param HAS_HUE_TEX 0 1
//#param HAS_HUEMASK_TEX 0 1
//#param HAS_GRUNGE_TEX 0 1
//#param IS_ETHEREAL 0 1

struct PS_INPUT
{
    float2 texCoord0		: TEXCOORD0;	// sprite uv
    float2 texCoord1		: TEXCOORD1;	// world space uv
    float2 texCoord2		: TEXCOORD2;	// offset into hue texture
	float4 color			: COLOR0;		// diffuse color
};

// TODO: have c++ inform the shader of these constants so we can change the size of the hue
// texture if needed.
static const float c_fHueTextureHeight = 1024.0f;
static const float c_fNumHuePixels = 255.0f;
static const float c_fHuePixelUVOffset = (1.0f / c_fHueTextureHeight) * c_fNumHuePixels;
static const half3 LuminanceConv = { 0.2125f, 0.7154f, 0.0721f };
static const half3 EtherealConv = { 0.0f, 0.01f, 0.025f };

half4 main( uniform sampler2D EffectSampler, 
			uniform sampler2D HueSampler, 
			uniform sampler2D HueMaskSampler, 
			uniform sampler2D GrungeSampler, 
			PS_INPUT IN ) : COLOR0
{
	// get main sprite texture pixel
	float4 colorout = tex2D(EffectSampler, IN.texCoord0);

	if ( colorout.a > 0.0f)
	{
		#if HAS_HUE_TEX == 1
			// hue effect if we have an index into the hue texture, 0,0 means not to hue
			if (IN.texCoord2.x > 0.0f || IN.texCoord2.y > 0.0f)
			{
				// index into hue by calculating the pixel's luminance
				float hueIndex = dot((half3)colorout, LuminanceConv) * c_fHuePixelUVOffset;
				float4 hueColor = tex2D(HueSampler, IN.texCoord2 + float2(hueIndex, 0.0f));
				colorout.rgb = hueColor.rgb;
				colorout.a *= hueColor.a; // multiply alpha to maintain the proper effect edge	
			}
		#endif //HAS_HUE_TEX
	}
	
	// modulate by diffuse alpha (no need to use diffuse RGB since the pixel is already hued
	colorout.a *= IN.color.a;
	return colorout;
}

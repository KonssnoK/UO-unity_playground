////Death effect shader
//#target ps_2_0

static const half3 LuminanceConv = { 0.2125f, 0.7154f, 0.0721f };

half4 Monochrome(in half4 color) : COLOR0
{
    return dot((half3)color, LuminanceConv);
}

struct PS_INPUT
{
    half4 texCoord0 : TEXCOORD0;
};

half4 main(uniform sampler2D ScreenSampler
            ,PS_INPUT IN) : COLOR0
{			
    half4 color = Monochrome(tex2D(ScreenSampler, IN.texCoord0));
    return color;
}
